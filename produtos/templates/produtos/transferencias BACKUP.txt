    <script>
        let sugestoesGlobais = [];

        document.addEventListener("DOMContentLoaded", () => {
            carregarDados();
        });

        async function carregarDados() {
            try {
                // Ajuste esta rota caso as suas views não estejam mapeadas em /estoque/...
                const response = await fetch('/estoque/api_sugestoes_transferencia/');
                const data = await response.json();
                
                if (data.erro) {
                    document.getElementById("tabela-corpo").innerHTML = `<tr><td colspan="7" class="p-10 text-center text-red-500 font-bold uppercase">ERRO DE CONEXÃO COM O ERP:<br><span class="text-xs text-slate-500 normal-case mt-2 block">${data.erro}</span></td></tr>`;
                    return;
                }

                if (data.ultima_atualizacao) {
                    const lblData = document.getElementById('data-ultima-atualizacao');
                    if(lblData) lblData.innerText = data.ultima_atualizacao;
                }

                sugestoesGlobais = data.sugestoes || [];
                renderizarTabela('TODOS');
            } catch (error) {
                console.error(error);
                document.getElementById("tabela-corpo").innerHTML = `<tr><td colspan="7" class="p-10 text-center text-red-500 font-bold">Erro ao conectar com servidor.</td></tr>`;
            }
        }

        function renderizarTabela(filtro) {
            const tbody = document.getElementById("tabela-corpo");
            tbody.innerHTML = "";
            document.getElementById("chk-todos").checked = false;

            const filtrados = sugestoesGlobais.filter(item => {
                if (filtro === 'TODOS') return true;
                if (filtro === 'TRANSFERIR' && item.status.includes('TRANSFERIR')) return true;
                if (filtro === 'COMPRAR' && item.status.includes('COMPRAR')) return true;
                if (filtro === 'PRIORIDADE' && item.capacidade_maxima == -1 && item.saldo_vila > 0) return true;
                if (filtro === 'SEPARANDO' && item.status === 'SEPARANDO') return true;
                if (filtro === 'TODOS' && item.status === 'MANUAL') return true;
                return false;
            });

            // Atualiza o visual dos botões de filtro na tela
            document.querySelectorAll('.filter-btn').forEach(btn => {
                const btnFiltro = btn.getAttribute('data-filtro');
                const activeCls = btn.getAttribute('data-active').split(' ');
                const inactiveCls = btn.getAttribute('data-inactive').split(' ');
                if (btnFiltro === filtro) {
                    btn.classList.remove(...inactiveCls);
                    btn.classList.add(...activeCls);
                } else {
                    btn.classList.remove(...activeCls);
                    btn.classList.add(...inactiveCls);
                }
            });

            if (filtrados.length === 0) {
                tbody.innerHTML = `<tr><td colspan="7" class="p-20 text-center font-black text-slate-200 uppercase italic text-2xl tracking-widest">Tudo OK! Nenhuma ação necessária.</td></tr>`;
                return;
            }

            // Prevenção de Congelamento: Mostra apenas os 300 primeiros na tela
            const limiteDOM = 300;
            const itensExibidos = filtrados.slice(0, limiteDOM);

            itensExibidos.forEach(item => {
                const tr = document.createElement("tr");
                tr.className = "hover:bg-slate-50/50 transition-colors esconder-na-impressao";
                const nomeEscapado = item.nome.replace(/'/g, "\\'"); // Evita erro no javascript ao clicar
                
                let corBadge = "bg-slate-100 text-slate-500";
                let badgeTexto = item.status.replace('_', ' + ');

                if (item.configurado) {
                    if(item.status === 'OK') corBadge = "bg-green-100 text-green-600 border border-green-200";
                    if(item.status === 'TRANSFERIR') corBadge = "bg-blue-100 text-blue-600 border border-blue-200";
                    if(item.status === 'TRANSFERIR_COMPRAR') corBadge = "bg-orange-100 text-orange-600 border border-orange-200";
                    if(item.status === 'COMPRAR') corBadge = "bg-red-100 text-red-600 border border-red-200";
                    if(item.status === 'SEPARANDO') corBadge = "bg-yellow-100 text-yellow-700 border border-yellow-300 animate-pulse";
                    if(item.status === 'MANUAL') { corBadge = "bg-purple-100 text-purple-600 border border-purple-200"; badgeTexto = "📦 MANUAL"; }
                } else {
                    if (item.status === 'ALTA') {
                        corBadge = "bg-red-600 text-white font-black shadow-md border-red-700 animate-pulse";
                        badgeTexto = "🔴 ALTA PRIORIDADE";
                    } else if (item.status === 'MEDIA') {
                        corBadge = "bg-amber-400 text-white font-black shadow-sm border-amber-500";
                        badgeTexto = "🟡 MÉDIA PRIORIDADE";
                    }
                    if(item.status === 'SEPARANDO') corBadge = "bg-yellow-100 text-yellow-700 border border-yellow-300 animate-pulse";
                    if(item.status === 'MANUAL') { corBadge = "bg-purple-100 text-purple-600 border border-purple-200"; badgeTexto = "📦 MANUAL"; }
                }

                // Somente deixa habilitado o checkbox se houver algo para transferir
                const chkBoxHTML = item.status !== 'SEPARANDO'
                    ? `<input type="checkbox" class="item-checkbox w-6 h-6 cursor-pointer accent-blue-600" data-id="${item.produto_id}" data-nome="${nomeEscapado}" data-barras="${item.codigo_barras || item.codigo}" ${item.qtde_transferir > 0 ? 'checked' : ''}>` 
                    : ``;

                let centroColHTML = '';
                if (item.configurado) {
                    let maxBadgeHTML = '';
                    if (item.capacidade_maxima == -1 && item.saldo_vila > 0) {
                        maxBadgeHTML = `<div class="text-[11px] text-amber-700 font-bold uppercase bg-amber-100 border border-amber-300 shadow-sm rounded-md px-1.5 py-0.5 cursor-help" title="PRIORIDADE: Há saldo na Vila Elias! Recomenda-se definir uma Capacidade Máxima para evitar transferir todo o estoque de uma vez.">
                            Máx: ∞ <span class="ml-0.5 text-xs">⚠️</span>
                        </div>`;
                    } else {
                        maxBadgeHTML = `<div class="text-[11px] text-slate-500 font-bold uppercase bg-slate-100 rounded-md px-1.5 py-0.5 cursor-help" title="MÁXIMO (Limite): Quantidade máxima que cabe no Centro. Se estiver com infinito (∞), o sistema vai sugerir transferir tudo que houver disponível na Vila Elias.">
                            Máx: ${item.capacidade_maxima == -1 ? '∞' : item.capacidade_maxima.toFixed(1)}<span class="text-blue-500 font-black ml-0.5">?</span>
                        </div>`;
                    }
                    centroColHTML = `
                        <div class="font-bold text-slate-600 text-xl">${item.saldo_centro.toFixed(1)}</div>
                        <div class="flex items-center justify-center gap-1 mt-1">
                            <div class="text-[11px] text-slate-500 font-bold uppercase bg-slate-100 rounded-md px-1.5 py-0.5 cursor-help" title="MÍNIMO (Gatilho): É a soma da Média de Vendas com o Estoque de Segurança. Quando o saldo do Centro chegar neste valor, o sistema pedirá transferência.">
                                Mín: ${item.capacidade_minima.toFixed(1)}<span class="text-blue-500 font-black ml-0.5">?</span>
                            </div>
                            ${maxBadgeHTML}
                        </div>`;
                } else {
                    centroColHTML = `
                        <div class="font-bold text-slate-400 text-xl opacity-50">${item.saldo_centro.toFixed(1)}</div>
                        <div class="mt-1">
                            <button onclick="editarRegras('${item.produto_id}', '${nomeEscapado}', '', '')" class="text-[11px] px-3 py-1 text-white font-bold uppercase bg-slate-400 hover:bg-slate-600 transition-colors rounded-md shadow-sm">⚙️ Configurar Regra</button>
                        </div>`;
                }

                let acoesHTML = `
                    <button onclick="abrirModalAjuste('${item.produto_id}', '${nomeEscapado}', ${item.saldo_centro}, ${item.saldo_vila}, '${item.codigo}')" class="text-amber-500 hover:text-amber-700 transition-colors mr-2 text-xl" title="Ajuste de Estoque">✏️</button>
                    <button onclick="editarRegras('${item.produto_id}', '${nomeEscapado}', ${item.capacidade_maxima || ''}, ${item.estoque_seguranca || ''})" class="text-slate-400 hover:text-slate-800 transition-colors text-xl" title="Editar Capacidade e Segurança">⚙️</button>
                `;
                if (item.status === 'SEPARANDO') {
                    acoesHTML = `<button onclick="cancelarSeparacao('${item.produto_id}')" class="text-red-500 hover:text-red-700 font-bold text-xs uppercase px-3 py-1.5 bg-red-50 rounded border border-red-200 mr-2" title="Cancelar Separação">❌ Cancelar</button>` + acoesHTML;
                }

                let qtdeHTML = '';
                if (item.status === 'SEPARANDO') {
                    qtdeHTML = `<span class="bg-yellow-50 text-yellow-700 px-4 py-1.5 rounded-lg font-black text-sm border border-yellow-200">TR: ${item.qtde_transferir.toFixed(0)}</span>`;
                } else {
                    qtdeHTML = `
                        <div class="flex items-center justify-end gap-1">
                            <span class="text-xs font-black text-slate-400">TR:</span>
                            <input type="number" min="0" step="1" class="qtde-input w-20 text-lg py-1 px-1 text-center font-black text-blue-600 bg-blue-50 border border-blue-200 rounded-lg outline-none focus:ring-2 focus:ring-blue-400" value="${item.qtde_transferir > 0 ? item.qtde_transferir.toFixed(0) : ''}" onchange="toggleCheckboxManual(this)">
                        </div>
                    `;
                }
                if (item.qtde_comprar > 0) {
                    qtdeHTML += `<span class="bg-red-50 text-red-600 px-3 py-1 rounded-lg font-black text-sm border border-red-100 mt-1 inline-block">CP: ${item.qtde_comprar.toFixed(0)}</span>`;
                }

                tr.innerHTML = `
                    <td class="p-5 text-center">${chkBoxHTML}</td>

        async function gerarCupom() {
            const selecionados = document.querySelectorAll('.item-checkbox:checked');
            if (selecionados.length === 0) {
                alert("Selecione pelo menos um produto para gerar a separação!");
                return;
            }

            const cupomDiv = document.getElementById("cupom-conteudo");
            const itensAPI = [];
            let selecionadosValidos = 0;
            let dataStr = new Date().toLocaleString('pt-BR');
            
            let html = `
                <div style="text-align:center; font-weight:bold; margin-bottom: 10px; font-size: 16px;">
                    SEPARAÇÃO - VILA ELIAS<br>
                    <span style="font-size: 11px; font-weight: normal;">${dataStr}</span>
                </div>
                <div style="border-bottom: 2px dashed #000; margin-bottom: 15px;"></div>
            `;

            selecionados.forEach(chk => {
                const id = chk.getAttribute("data-id");
                const nome = chk.getAttribute("data-nome");
                const barras = chk.getAttribute("data-barras");
                const tr = chk.closest('tr');
                const inputQtde = tr.querySelector('.qtde-input');
                const qtde = inputQtde && inputQtde.value ? parseFloat(inputQtde.value) : 0;
                
                if (qtde > 0) {
                    selecionadosValidos++;
                    let barcodeHTML = '';
                    if (barras && barras.trim() !== '' && barras !== 'undefined') {
                        barcodeHTML = `<div style="text-align: left; margin-left: 15px; margin-top: 5px; margin-bottom: 5px;"><svg class="barcode-svg" jsbarcode-value="${barras}" jsbarcode-height="30" jsbarcode-width="1.5" jsbarcode-fontSize="12" jsbarcode-margin="0"></svg></div>`;
                    }

                    itensAPI.push({ id: id, qtde: qtde });

                    html += `
                        <div class="linha-produto">
                            <div style="font-weight: bold; font-size: 15px;">[  ] ${qtde}x</div>
                            <div style="font-size: 13px; margin-left: 20px; text-transform: uppercase;">${nome}</div>
                            ${barcodeHTML}
                        </div>
                    `;
                }
            });

            if (selecionadosValidos === 0) {
                alert("Nenhuma quantidade válida informada para transferência!");
                return;
            }

            html += `
                <div style="border-bottom: 2px dashed #000; margin-top: 15px; margin-bottom: 10px;"></div>
                <div style="text-align:center; font-size: 12px; font-weight: bold;">Total de volumes: ${selecionadosValidos}</div>
            `;

            cupomDiv.innerHTML = html;
            
            try {
                await fetch('/estoque/api_registrar_impressao/', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'X-CSRFToken': '{{ csrf_token }}' },
                    body: JSON.stringify({ itens: itensAPI })
                });
            } catch (e) {}

            // Inicia o desenho dos códigos de barra reais na tela oculta
            try { JsBarcode(".barcode-svg").init(); } catch(e) { console.log(e); }
            
            window.print(); // Dispara o diálogo de impressão focado na div via CSS
            setTimeout(() => carregarDados(), 2000);
        }

        // --- SISTEMA DE MODAIS ---
        let modalPromiseResolve = null;
        let depositoSelecionadoParaAjuste = '';

        function abrirModal(id) {
            document.getElementById('backdrop').classList.remove('hidden');
            document.getElementById(id).classList.remove('hidden');
            setTimeout(() => {
                document.getElementById('backdrop').classList.add('backdrop-enter');
                document.getElementById(id).classList.add('modal-enter');
                const firstInput = document.getElementById(id).querySelector('input');
                if(firstInput) firstInput.focus();
            }, 10);
        }

        function fecharModal(id, confirmado) {
            document.getElementById('backdrop').classList.remove('backdrop-enter');
            document.getElementById(id).classList.remove('modal-enter');
            setTimeout(() => {
                document.getElementById('backdrop').classList.add('hidden');
                document.getElementById(id).classList.add('hidden');
                if (modalPromiseResolve) {
                    modalPromiseResolve(confirmado);
                    modalPromiseResolve = null;
                }
            }, 300);
        }

        async function carregarUsuariosDropdown() {
            const select = document.getElementById('select-usuario');
            select.innerHTML = '<option value="">Carregando usuários...</option>';
            try {
                const res = await fetch('/estoque/api_listar_usuarios/');
                const data = await res.json();
                if (data.ok) {
                    select.innerHTML = '<option value="">-- Selecione seu nome --</option>';
                    data.usuarios.forEach(u => {
                        select.innerHTML += `<option value="${u.id}">${u.nome}</option>`;
                    });
                } else {
                    select.innerHTML = '<option value="">Erro ao carregar</option>';
                }
            } catch(e) { select.innerHTML = '<option value="">Erro de conexão</option>'; }
        }

        async function pedirPIN(mensagem = "Confirme com seu PIN") {
            document.getElementById('pin-desc').innerText = mensagem;
            document.getElementById('input-pin').value = '';
            abrirModal('modal-pin');
            
            const confirmado = await new Promise(resolve => { modalPromiseResolve = resolve; });
            if (!confirmado) return null;

            const pin = document.getElementById('input-pin').value.trim();
            if (!pin) return null;

            if (pin === '1234') {
                document.getElementById('input-novo-pin').value = '';
                carregarUsuariosDropdown();
                abrirModal('modal-troca-pin');
                throw new Error("Troca de PIN necessária");
            }
            return pin;
        }

        async function salvarNovoPIN() {
            const perfilId = document.getElementById('select-usuario').value;
            if (!perfilId) { alert("Por favor, selecione o seu nome na lista!"); return; }

            const novoPin = document.getElementById('input-novo-pin').value.trim();
            if (novoPin.length < 4) { alert("O PIN deve ter no mínimo 4 dígitos."); return; }
            if (novoPin === '1234') { alert("Você não pode usar 1234. Escolha outro!"); return; }

            const fd = new FormData();
            fd.append('perfil_id', perfilId);
            fd.append('pin_atual', '1234');
            fd.append('novo_pin', novoPin);
            fd.append('csrfmiddlewaretoken', '{{ csrf_token }}');

            const res = await fetch('/estoque/api_atualizar_pin/', { method: 'POST', body: fd });
            const data = await res.json();
            if (data.ok) {
                alert("✅ PIN Atualizado! Refaça a operação agora usando seu novo PIN.");
                fecharModal('modal-troca-pin', false);
            } else { alert(data.erro); }
        }

        async function editarRegras(produtoId, nomeProduto, atualMax = '', atualSeg = '') {
            document.getElementById('regra-produto-nome').innerText = nomeProduto;
            document.getElementById('input-regra-max').value = atualMax == -1 ? '' : atualMax;
            document.getElementById('input-regra-seg').value = atualSeg;
            
            abrirModal('modal-regra');
            const confirmado = await new Promise(resolve => { modalPromiseResolve = resolve; });
            if (!confirmado) return;

            const p_max = document.getElementById('input-regra-max').value;
            const p_seg = document.getElementById('input-regra-seg').value;
            if (p_seg.trim() === '') { alert("Estoque Segurança é obrigatório."); return; }

            try {
                const pin = await pedirPIN("Confirme para salvar as regras");
                if (!pin) return;

                const fd = new FormData();
                fd.append('produto_id', produtoId);
                fd.append('nome_produto', nomeProduto);
                fd.append('capacidade_maxima', p_max.trim() === '' ? '-1' : p_max);
                fd.append('estoque_seguranca', p_seg);
                fd.append('pin', pin);
                fd.append('csrfmiddlewaretoken', '{{ csrf_token }}');

                const res = await fetch('/estoque/api_salvar_config_transferencia/', { method: 'POST', body: fd });
                const data = await res.json();
                if (data.ok) carregarDados();
                else alert("Erro ao salvar: " + data.erro);
            } catch (e) { if(e.message !== "Troca de PIN necessária") alert("Erro de comunicação."); }
        }

        async function adicionarRegraNova() {
            const termo = prompt("🔍 Busca Limpa:\nDigite parte do NOME ou CÓDIGO do produto que deseja adicionar:");
            if (!termo || termo.trim() === '') return;

            try {
                const response = await fetch(`/api/buscar/?q=${encodeURIComponent(termo)}`);
                const data = await response.json();
                
                if (!data.produtos || data.produtos.length === 0) {
                    alert("Nenhum produto encontrado com esse termo."); return;
                }
                const p = data.produtos[0];
                editarRegras(p.id, p.nome, '', '');
            } catch (error) { alert("Erro ao buscar produto."); }
        }

        async function adicionarTransferenciaManual() {
            const termo = prompt("📦 Transferência Manual:\nDigite o NOME ou CÓDIGO do produto que deseja transferir:");
            if (!termo || termo.trim() === '') return;

            try {
                const response = await fetch(`/api/buscar/?q=${encodeURIComponent(termo)}`);
                const data = await response.json();
                
                if (!data.produtos || data.produtos.length === 0) {
                    alert("Nenhum produto encontrado."); return;
                }
                const p = data.produtos[0];
                
                const existe = sugestoesGlobais.find(i => i.produto_id === p.id);
                if (!existe) {
                    sugestoesGlobais.unshift({
                        produto_id: p.id, codigo: p.codigo_nfe || p.codigo_interno, codigo_barras: p.codigo_barras,
                        nome: p.nome, saldo_centro: p.saldo_centro, saldo_vila: p.saldo_vila,
                        status: "MANUAL", qtde_transferir: 1, qtde_comprar: 0,
                        configurado: false, prioridade: 0
                    });
                } else {
                    existe.qtde_transferir = 1;
                    if(existe.status !== 'SEPARANDO') existe.status = "MANUAL";
                    existe.prioridade = 0;
                    sugestoesGlobais.sort((a,b) => a.prioridade - b.prioridade);
                }
                renderizarTabela('TODOS');
            } catch (error) { alert("Erro ao buscar produto."); }
        }

                fd.append('pin', pin);
                fd.append('csrfmiddlewaretoken', '{{ csrf_token }}');

                const res = await fetch('/estoque/api_importar_planilha/', { method: 'POST', body: fd });

                if (data.ok) {
                    alert(data.mensagem);
                    carregarDados();

        }

        // --- ATALHOS DE TECLADO ---
        document.addEventListener('keydown', function(e) {
            // Se houver algum modal aberto, permitimos apenas o uso do ESC para fechar as caixinhas
            if (!document.getElementById('backdrop').classList.contains('hidden')) {
                if (e.key === 'Escape') {
                    ['modal-pin', 'modal-troca-pin', 'modal-regra', 'modal-ajuste'].forEach(id => {
                        if (!document.getElementById(id).classList.contains('hidden')) {
                            fecharModal(id, false);
                        }
                    });
                }
                return;
            }

            if (e.key === 'F2') {
                e.preventDefault(); adicionarRegraNova();
            } else if (e.key === 'F3') {
                e.preventDefault(); adicionarTransferenciaManual();
            } else if (e.key === 'F4') {
                e.preventDefault(); renderizarTabela('TODOS');
            } else if (e.key === 'F5') {
                e.preventDefault(); atualizarMedias();
            } else if (e.key === 'F7') {
                e.preventDefault(); gerarCupom();
            } else if (e.key === 'F8') {
                e.preventDefault(); document.getElementById('arquivo-csv').click();
            }
        });
